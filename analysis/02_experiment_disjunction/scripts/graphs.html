Looing at RT1

Overall raw RT 1

d.raw.rt1$ReactionTime1 <- as.numeric(d.raw.rt1$ReactionTime1)

agr <- d.raw.rt1 %>%
  filter(Connective %in% c("and","but","so"))%>% 
  group_by(Connective) %>%
  summarize(meanRT1 = mean(ReactionTime1), CILow = ci.low(ReactionTime1), CIHigh = ci.high(ReactionTime1)) %>% 
  mutate(YMin = meanRT1 - CILow, YMax = meanRT1 + CIHigh)
## `summarise()` ungrouping output (override with `.groups` argument)
dodge = position_dodge(.9)
ggplot(agr,aes(x=Connective, y=meanRT1, fill=Connective)) +
    geom_bar(position=dodge,stat="identity") +
    geom_errorbar(aes(ymin=YMin,ymax=YMax),width=.25,position=dodge)

Look at order

agr <- d.raw.rt1 %>%
  filter(Connective %in% c("and","but","so"))%>% 
  group_by(Connective,Order) %>%
  summarize(meanRT1 = mean(ReactionTime1), CILow = ci.low(ReactionTime1), CIHigh = ci.high(ReactionTime1)) %>% 
  mutate(YMin = meanRT1 - CILow, YMax = meanRT1 + CIHigh)
## `summarise()` regrouping output by 'Connective' (override with `.groups` argument)
dodge = position_dodge(.9)
ggplot(agr,aes(x=Connective, y=meanRT1, fill=Connective)) +
    facet_wrap(~Order) +
    geom_bar(position=dodge,stat="identity") +
    geom_errorbar(aes(ymin=YMin,ymax=YMax),width=.25,position=dodge)
